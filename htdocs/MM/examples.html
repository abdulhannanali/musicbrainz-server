Simple audio track example

<?xml version="1.0"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#" 
         xmlns:MM = "http://cdindex.org/MM#" >

  <rdf:Description 
       about = "http://www.cdindex.org/cgi-bin/cdi/xget.pl?albumid=3">
    <DC:Title>Strangers</DC:Title>
    <DC:Creator>Portishead</DC:Creator>
    <MM:Album>Dummy</MM:Album>
    <DC:Relation track="3"/>
    <DC:Format duration="195000"/>
    <DC:Description remixInfo="Airbus Reconstruction"
                    recordedAt="Performed live at CCH 3, Hamburg"/>
    <DC:Contributor role="Performer" 
                    roleDetail="Lead Vocals">Beth Gibbons</DC:Contributor>
  </rdf:Description>
</rdf:RDF>

Audio track with sync lyrics example

<?xml version="1.0"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#" 
         xmlns:MM = "http://cdindex.org/MM#" >

  <rdf:Description about = 
       "http://www.cdindex.org/cgi-bin/cdi/xget.pl?albumid=512">
    <DC:Title>Big Time</DC:Title>
    <DC:Creator>Peter Gabriel</DC:Creator>
    <MM:Album>So</MM:Album>
    <DC:Relation track="7"/>
    <MM:SyncEvents>
       <rdf:Seq>
          <rdf:li>
             <rdf:Description>
                <MM:SyncText ts="1400">Hi There!</MM:SyncText>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <MM:SyncText ts="2100">I'm on my way</MM:SyncText>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <MM:SyncText ts="2870">I'm making it</MM:SyncText>
             </rdf:Description>
          </rdf:li>
       </rdf:Bag>
    </MM:SyncEvents>
  </rdf:Description>
</rdf:RDF>

Classical audio track example

<?xml version="1.0"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#" 
         xmlns:MM = "http://cdindex.org/MM#" >

  <rdf:Description about=
       "http://www.cdindex.org/cgi-bin/cdi/xget.pl?albumid=7664">
    <DC:Title>Tristan Und Isolde</DC:Title>
    <DC:Description primaryKey="B Major" 
           recordedAt="Recorded at Theatre Royal, Glasgow, Scotland"
           memberOfSubpart="Act II"/>
    <MM:Contributors>
       <rdf:Bag>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Composer">Richard Wagner</DC:Contributor>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Conductor">Richard Armstrong</DC:Contributor>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Orchestra" >Scottish Opera</DC:Contributor>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Performer" 
                     roleDetail="Tenor">John Murray</DC:Contributor>
             </rdf:Description>
          </rdf:li>
       </rdf:Bag>
    </MM:Contributors>
    <DC:Relation partsInSet="5" isPartOf="2"/>
  </rdf:Description>
</rdf:RDF>

Video track example

<?xml version="1.0"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#" 
         xmlns:MM = "http://cdindex.org/MM#" >

  <rdf:Description about = "http://us.imdb.com/Title?0169547" >
    <DC:Title>American Beauty</DC:Title>

    <DC:Date issued="1999-06-08"/>
    <MM:Contributors>
       <rdf:Bag>
          <rdf:li>

             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Actor" 
                                roleDetail="Lester Burnham">Kevin Spacey</DC:Contributor>
             </rdf:Description>

             <rdf:Description>
                <DC:Contributor role="Actor" roleDetail="Angela Hayes">Mena Suvari</DC:Contributor>

  </rdf:Description>

Simple audio track example

<?xml version="1.0" encoding="ISO-8859-1"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#"
         xmlns:MM = "http://musicbrainz.org/MM#">

  <rdf:Description>
    <DC:Title>Strangers</DC:Title>
    <DC:Creator>Portishead</DC:Creator>
    <DC:Identifier artistId="00003C7D@fuss@39A2D955"/>
    <MM:TrackNum>5</MM:TrackNum>
    <DC:Relation type="album">
      <rdf:Description>
        <DC:Title>Dummy</DC:Title>
        <DC:Identifier albumId="00003C7E@fuss@39A2D955"/>
      </rdf:Description>
    </DC:Relation>
  </rdf:Description>
</rdf:RDF>

Audio track with sync lyrics example

<?xml version="1.0"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#" 
         xmlns:MM = "http://musicbrainz.org/MM#" >

  <rdf:Description>
    <DC:Title>Big Time</DC:Title>
    <DC:Creator>Peter Gabriel</DC:Creator>
    <DC:Relation type="album">So</MM:Album>
      <rdf:Description>
        <DC:Title>So</DC:Title>
        <DC:Identifier albumId="000036F4@fatman.freeamp.org@39CCBC7E"/>
      </rdf:Description>
    </DC:Relation>
    <MM:TrackNum track="7"/>
    <MM:SyncEvents>
       <rdf:Seq>
          <rdf:li>
             <rdf:Description>
                <MM:SyncText ts="1400">Hi There!</MM:SyncText>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <MM:SyncText ts="2100">I'm on my way</MM:SyncText>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <MM:SyncText ts="2870">I'm making it</MM:SyncText>
             </rdf:Description>
          </rdf:li>
       </rdf:Bag>
    </MM:SyncEvents>
  </rdf:Description>
</rdf:RDF>

Classical audio track example

<?xml version="1.0"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#" 
         xmlns:MM = "http://musicbrainz.org/MM#" >

  <rdf:Description>
    <DC:Title>Tristan Und Isolde</DC:Title>
    <DC:Description primaryKey="B Major" 
           recordedAt="Recorded at Theatre Royal, Glasgow, Scotland"
           memberOfSubpart="Act II"/>
    <MM:Contributors>
       <rdf:Bag>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Composer">Richard Wagner</DC:Contributor>>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Conductor">Richard Armstrong</DC:Contributor>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Orchestra" >Scottish Opera</DC:Contributor>
             </rdf:Description>
          </rdf:li>
          <rdf:li>
             <rdf:Description>
                <DC:Contributor role="Performer" 
                     roleDetail="Tenor">John Murray</DC:Contributor>
             </rdf:Description>
          </rdf:li>
       </rdf:Bag>
    </MM:Contributors>
    <DC:Relation partsInSet="5" isPartOf="2"/>
  </rdf:Description>
</rdf:RDF>

Video track example

<?xml version="1.0"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#"

  <rdf:Description about = "http://us.imdb.com/Title?0169547" >

    <DC:Date issued="1999-06-08"/>

             </rdf:Description>
          </rdf:li>
          <rdf:li>

Album example

<?xml version="1.0" encoding="ISO-8859-1"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#"
         xmlns:MM = "http://musicbrainz.org/MM#">

  <rdf:Description>
    <MM:Collection type="album" numParts="1">
      <rdf:Description>
        <DC:Title>Corroded Path</DC:Title>
        <DC:Creator>Network</DC:Creator>
        <DC:Identifier albumId="00003C7E@fuss@39A2D955"
                       artistId="00003C7D@fuss@39A2D955"/>
        <rdf:Seq>
          <rdf:li>
            <rdf:Description>
              <DC:Title>Hyperventilation</DC:Title>
              <DC:Identifier trackId="00003C7F@fuss@39A2D955"/>
              <MM:TrackNum>1</MM:TrackNum>
            </rdf:Description>
          </rdf:li>
        </rdf:Seq>
      </rdf:Description>
    </MM:Collection>
  </rdf:Description>
</rdf:RDF>

Playlist example

<?xml version="1.0" encoding="ISO-8859-1"?>
<rdf:RDF xmlns = "http://w3.org/TR/1999/PR-rdf-syntax-19990105#"
         xmlns:DC = "http://purl.org/DC#"
         xmlns:MM = "http://musicbrainz.org/MM#">

  <rdf:Description>
    <MM:Collection type="playlist" numParts="8">
      <rdf:Description>
        <DC:Title>Robert's Favorite Playlist</DC:Title>
        <DC:Creator>Robert Kaye</DC:Creator>
        <rdf:Seq>
          <rdf:li>
            <rdf:Description>
              <DC:Identifier artistId="00003C7D@fuss@39A2D955"
                             trackId="00003C7F@fuss@39A2D955"/>
              <DC:Title>Hyperventilation</DC:Title>
              <DC:Creator>Network</DC:Creator>
              <DC:Relation type="album">
                <rdf:Description>
                   <DC:Creator>Network</DC:Creator>
                   <DC:Identifier albumId="00003C7E@fuss@39A2D955"/>
                </rdf:Description>
              </DC:Relation>
            </rdf:Description>
          </rdf:li>
        </rdf:Seq>
      </rdf:Description>
    </MM:Collection>
  </rdf:Description>
</rdf:RDF>
